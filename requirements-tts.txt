# =============================================================================
# Hornelore — TTS venv requirements
# =============================================================================
# WO-11: Standalone Hornelore repo. This file pins every dependency to the
# exact version that was running in the old lorevox/.venv-tts when WO-10M
# was verified green.
#
# Target: Python 3.12, WSL2 Ubuntu. CPU inference by default (TTS_GPU=0 in
# .env keeps VRAM free for the LLM + STT stack on the GPU venv).
#
# -----------------------------------------------------------------------------
# INSTALL ORDER
# -----------------------------------------------------------------------------
#   cd /mnt/c/Users/chris/hornelore
#   python3 -m venv .venv-tts
#   source .venv-tts/bin/activate
#   pip install -r requirements-tts.txt
#
#   # Pre-download the TTS model so first startup is fast:
#   python -c "from TTS.api import TTS; TTS('tts_models/en/vctk/vits')"
#
# -----------------------------------------------------------------------------
# NOTES
# -----------------------------------------------------------------------------
# - Uses the `coqui-tts` fork on PyPI, NOT the deprecated `TTS` package.
#   The Python import is still `from TTS.api import TTS`.
# - torch 2.8.0 stable is fine for the TTS venv — the TTS workload does not
#   need Blackwell sm_100 kernels (runs on CPU in .env default).
# - nvidia_* packages are omitted — torch pulls them transitively.
# =============================================================================
