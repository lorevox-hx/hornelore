# Target: Python 3.12, WSL2 Ubuntu, RTX 5080 Laptop (Blackwell sm_100).
#
# -----------------------------------------------------------------------------
# INSTALL ORDER
# -----------------------------------------------------------------------------
#   cd /mnt/c/Users/chris/hornelore
#   python3 -m venv .venv-gpu
#   source .venv-gpu/bin/activate
#
#   # 1) PyTorch nightly for Blackwell (sm_100, CUDA 12.8). MUST come first.
#   pip install --pre torch torchvision torchaudio \
#     --index-url https://download.pytorch.org/whl/nightly/cu128
#
#   # 2) Everything else pinned below.
#   pip install -r requirements-gpu.txt
#
#   # 3) Sanity check.
#   python -c "import torch, transformers, bitsandbytes; print(torch.cuda.get_device_name(0))"
#   # Expected: NVIDIA GeForce RTX 5080 Laptop GPU
#
# -----------------------------------------------------------------------------
# NOTES
# -----------------------------------------------------------------------------
# - torch / torchvision / torchaudio are intentionally OMITTED from this file
#   because they come from the nightly index, not PyPI. Installing them via
#   -r would pull stable wheels and break Blackwell support.
# - nvidia_* packages are also omitted — torch pulls them transitively.
# - xformers and triton are omitted for the same reason (torch wheel bundles).
# - The old lorevox venv contained a lot of packages hornelore does NOT need
#   (chromadb, sentence-transformers, diffusers, faiss, pandas, pyarrow,
#   unsloth, trl, cut-cross-entropy, ctranslate2, pymupdf, pytesseract,
#   ebooklib, pypandoc, datasets, torchao, opentelemetry, kubernetes, av,
#   beautifulsoup4, onnxruntime, grpcio, protobuf, bcrypt, etc.). Those
#   are lorevox-side and intentionally excluded from this hornelore-only
#   install to keep the venv lean.
# =============================================================================

# ── Photo system (services/photo_intake/*, services/photos/*) ──────────────
# Pillow drives BOTH thumbnail generation (services/photo_intake/thumbnail.py)
# AND EXIF auto-fill (services/photo_intake/exif.py). Both modules fail-soft
# on missing PIL — uploads succeed, thumbnails just come back as null +
# EXIF returns empty-shape — which produces a working-looking system that
# silently emits broken thumbnails. This bit us 2026-04-25 night and ate
# ~90 minutes of diagnosis. See docs/PILLOW-VENV-INSTALL.md for the
# diagnostic + verification path. Pinned to a recent version that
# handles modern Pixel/Samsung JPEG EXIF blocks reliably.
#
# Pillow ALSO drives the Document Archive lane (WO-MEDIA-ARCHIVE-01)
# for image-source items (scanned single-page docs, contact sheets,
# etc.) — same import path, same fail-soft behavior.
Pillow==12.2.0

# ── Document Archive (services/media_archive/*) ────────────────────────────
# pypdf — pure-Python PDF reader used for page_count detection on upload.
# Failure mode if missing: services/media_archive/text_probe.py falls back
# to page_count=NULL (visible in the curator list as missing pages-pill).
# Functionality not blocked, just degraded. Cheap to install, no system deps.
pypdf==6.4.1

# pdf2image — wraps poppler-utils (system dep) to render PDF first page as
# a JPEG thumbnail. Failure mode if poppler-utils missing: thumbnail step
# raises inside services/media_archive/thumbnail.py, which is wrapped in a
# try/except at the upload site, so the curator sees an inline SVG file-icon
# placeholder instead of a real thumb. Install poppler-utils via:
#   sudo apt install -y poppler-utils
# (Ubuntu/WSL2). Without poppler-utils the pip install still succeeds; the
# thumbnail just falls back to the placeholder, which is fine for working
# triage workflows. PDF page-count detection (pypdf above) is independent
# of poppler and works regardless.
pdf2image==1.17.0
